#Job Submission
sbatch jobscript.slurm

#查看队列：
squeue

#View job information
squeue -j jobid

#查看单个的作业情况：
squeue -j 1073460

# 查看自己的作业情况 注意whoami前后不是单引号
squeue -u xiahuang
squeue -u yihanzhou23

# 查看作业最新进程(必须在slurm-yourjobid.out所在的目录执行）
tail -F slurm-yourjobid.out

# 取消作业ID为123的作业
scancel 123

# 注意whoami前后不是单引号
scancel -u `whoami`

#增加可执行（x）权限
chmod +x ./template-surm.sh
#去除可执行（x）权限
chmod -x ./template-surm.sh

#查看权限
ls -l
